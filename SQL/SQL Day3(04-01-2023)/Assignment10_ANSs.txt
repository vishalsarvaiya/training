Tables Used:
===========
	Table - I :	officestaff
	Fields	:	recid (Primary Key), name, designation, salary, overtimesalary, bossno, addressid (Reference Key)


	Table - II :	address
	Fields :	recid (Primary Key), address1, address2, city, county, postalcode, country

CREATE TABLE trainee.Vishalofficestaff(
  recid INTEGER UNSIGNED primary key,
  name CHAR(25) NOT NULL,
  designation VARCHAR(25) not null,
  salary NUMERIC(6) NOT NULL,
  overtimesalary NUMERIC(4) NOT NULL,
  bossno VARCHAR(5) NOT NULL,
  addressid VARCHAR(5) NOT NULL
);


Write queries for following requirements:

Q-1	List Name, Designation and Salary of all "officestaff" members.
select name,designation,salary from Vishalofficestaff;

Q-2	List Name and Salary of all the people who are working as "Programmer".
select name,salary from Vishalofficestaff
where designation='programmer';

Q-3	List all the information about those people who are earning less than Rs.10,000/- .
select * from Vishalofficestaff
where salary<10000;

Q-4	List all the information about those staff members who fall in the salary range of 	Rs.5000/- to Rs.10,000/- .
select * from Vishalofficestaff
where salary BETWEEN 5000 AND 10000;

Q-5	List Name of all the office staff members in Lower Case.
select lower(name) from Vishalofficestaff;

Q-6	List Name of all the office staff members in following format:
	For example, if name is "umeshbhai" then display it in the format "Umeshbhai".
select concat(UPPER(SUBSTRING(name,1,1)),LOWER(SUBSTRING(name,2))) as "name" from Vishalofficestaff;

Q-7	List all the information about those people who works for "Nileshbhai".

select * from Vishalofficestaff
where bossno IN
(select bossno from Vishalofficestaff
where name='nileshbhai');


Q-8	List all the information about those members with salary greater than or equal to Rs.3000/-
 or working directly under "Nileshbhai".


select * from Vishalofficestaff
where salary>=3000 OR bossno IN
(select bossno from Vishalofficestaff
where name='nileshbhai');



Q-9	List Name, Salary, Over-time Salary of all the office staff members.

select name,salary,overtimesalary from Vishalofficestaff;


Q-10	Perform above MySQL statement again, but now 0(Zero) should be substituted for NULL in 	Over-time Salary field.

select name,salary,coalesce(overtimesalary,0)as "overtimesalary" from Vishalofficestaff;

Q-11	List Name and Total Monthly Salary (Salary + Over-time Salary) of all the office staff 	members.
select name, (salary+overtimesalary) as "Total Monthly Salary"
from Vishalofficestaff;


Q-12	List Name and Total Annual Salary of all the office staff members.

select name, ((salary+overtimesalary)*12) as "Total Annual Salary"
from Vishalofficestaff;

Q-13	List Name and Total Daily Salary of all the office staff members. (Consider a month is 	of 30 days).

select name, ((salary+overtimesalary)/30) as "Total Daily Salary"
from Vishalofficestaff;

Q-14	List all the information of the person earning Highest Salary (Salary + Over-time 	Salary) in the staff.


select name, salary+ifnull(overtimesalary,0) as "Highest Salary" from Vishalofficestaff
where salary+ifnull(overtimesalary,0)>= (select max(salary+ifnull(overtimesalary,0)) from Vishalofficestaff);



Q-15	List all the information of the person earning Least Salary (Salary + Over-time Salary) in the staff.
select name, salary+ifnull(overtimesalary,0) as "Least Salary" from Vishalofficestaff
where salary+ifnull(overtimesalary,0)<= (select min(salary+ifnull(overtimesalary,0)) from Vishalofficestaff);


Q-16	List all the information about those who earn the Highest Salary (Salary + Over-time 	Salary)
and those who earn the Least Salary (Salary + Over-time Salary) in the staff.

select name,recid, salary+ifnull(overtimesalary,0) as "Highest Salary"
from Vishalofficestaff
where salary+ifnull(overtimesalary,0)>= (select max(salary+ifnull(overtimesalary,0)) from Vishalofficestaff)
union
select name, recid, min(salary+ifnull(overtimesalary,0)) as "Least Salary"
from Vishalofficestaff
where salary+ifnull(overtimesalary,0)<= (select min(salary+ifnull(overtimesalary,0)) from Vishalofficestaff);


Q-17	List all the information about the person who earns more than his boss.
select *, max(salary+overtimesalary) as "Highest Salary"
from Vishalofficestaff
where salary>
(select salary from Vishalofficestaff
where designation='boss');


Q-18	List Name, Designation, Salary and City of each staff member.
select name,designation,salary,city
from Vishalofficestaff inner join Vishaladdress
on Vishalofficestaff.recid=Vishaladdress.recid;

Q-19	Using "address" table list total number of persons residing in "Ahmedabad" city.
select count(recid)
from Vishaladdress
where city='ahmedabad';


Q-20	List those City Name where 3 or more than 3 people resides.
select city
from Vishaladdress
group by city
having count(city)>=3;
